terminal IGUAL, DIGITO, SUMA, RESTA, DIVI, MULTI, POTENCIA, NEGATIVO, PDER, PIZQ;
non terminal sesion, ecuacion, expr, opunaria, opbinaria;

precedence left SUMA, RESTA;
precedence left MULTI, DIVI;
precedence left NEGATIVO;
precedence right POTENCIA;


sesion ::=     ecuacion 
             | ecuacion sesion
             ;
             
ecuacion ::=  expr {:  System.out.println("Ecuacion OK"); :} IGUAL;

expr ::=   DIGITO
         | expr opbinaria expr
         | opunaria expr
         | PIZQ expr PDER
         ;



opunaria ::= RESTA
            %prec NEGATIVO
             ;

opbinaria ::=   SUMA
              | RESTA
              | DIVI
              | MULTI
              | POTENCIA
              ;
